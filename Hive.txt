//----------------------------Customer_Hive------------------------//

drop table temp_Customer_external;
create external table temp_Customer_external (
  CUST_SSN INT,
  CUST_FIRST_NAME STRING,
  CUST_MIDDLE_NAME STRING,
  CUST_LAST_NAME STRING,
  CUST_CC_NO STRING,
  CUST_STREET STRING,
  CUST_CITY STRING,
  CUST_STATE STRING,
  CUST_COUNTRY STRING,
  CUST_ZIP INT,
  CUST_PHONE STRING,
  CUST_EMAIL STRING,
  LAST_UPDATED STRING)
ROW FORMAT DELIMITED
FIELDS TERMINATED BY ","
Stored as textfile
location "/user/maria_dev/Credit_Card_System/Customer_external";
load data inpath "/user/maria_dev/Credit_Card_System/Customer" OVERWRITE
into table temp_Customer_external;

drop table Customer;
create table Customer(
  CUST_SSN INT,
  CUST_FIRST_NAME STRING,
  CUST_MIDDLE_NAME STRING,
  CUST_LAST_NAME STRING,
  CUST_CC_NO STRING,
  CUST_STREET STRING,
  CUST_CITY STRING,
  CUST_COUNTRY STRING,
  CUST_ZIP INT,
  CUST_PHONE STRING,
  CUST_EMAIL STRING,
  LAST_UPDATED STRING)
  PARTITIONED BY (CUST_STATE STRING )
ROW FORMAT DELIMITED
FIELDS TERMINATED BY ","
Stored as textfile
location '/user/maria_dev/Credit_Card_System/Customer';

set hive.exec.dynamic.partition.mode=nonstrict;

INSERT INTO TABLE Customer PARTITION (CUST_STATE) 
SELECT 
  CUST_SSN,
  CUST_FIRST_NAME,
  CUST_MIDDLE_NAME,
  CUST_LAST_NAME,
  CUST_CC_NO,
  CUST_STREET,
  CUST_CITY,
  CUST_COUNTRY,
  CUST_ZIP,
  CUST_PHONE,
  CUST_EMAIL,
  LAST_UPDATED,
  CUST_STATE
FROM temp_Customer_external;

//----------------------------End_Customer_Hive------------------------//
